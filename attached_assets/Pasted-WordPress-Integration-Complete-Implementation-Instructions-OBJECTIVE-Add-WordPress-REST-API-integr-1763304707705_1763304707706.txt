SAVE TO WORDPRESS FUNCTION
Implement this function:
javascriptasync function savePromptToWordPress(promptTitle, promptContent, templateMode) {
  const wpUsername = process.env.WP_USERNAME;
  const wpAppPassword = process.env.WP_APP_PASSWORD;
  
  if (!wpUsername || !wpAppPassword) {
    throw new Error('WordPress credentials not configured');
  }
  
  const metadata = TEMPLATE_METADATA[templateMode];
  if (!metadata) {
    throw new Error(`Unknown template mode: ${templateMode}`);
  }
  
  const requestBody = {
    title: promptTitle,
    content: promptContent,
    status: 'publish',
    'prompt-category': [metadata.category],
    'use-case': metadata.useCases,
    'ai-model': metadata.aiModels
  };
  
  try {
    const response = await fetch('https://aifirstmovers.net/wp-json/wp/v2/prompts', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': 'Basic ' + btoa(`${wpUsername}:${wpAppPassword}`)
      },
      body: JSON.stringify(requestBody)
    });
    
    const data = await response.json();
    
    if (!response.ok) {
      throw new Error(data.message || `HTTP ${response.status}`);
    }
    
    return {
      success: true,
      id: data.id,
      url: data.link,
      title: data.title.rendered
    };
    
  } catch (error) {
    console.error('WordPress API Error:', error);
    return {
      success: false,
      error: error.message
    };
  }
}
